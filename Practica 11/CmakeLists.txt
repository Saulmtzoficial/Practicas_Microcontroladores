cmake_minimum_required(VERSION 3.13)

set(PICO_BOARD pico_w)
include(pico_sdk_import.cmake)

project(Practica_11)
pico_sdk_init()

add_executable(Practica11
    Practica11.c
)

pico_enable_stdio_usb(Practica11 1)
pico_enable_stdio_uart(Practica11 0)

target_link_libraries(Practica11
    pico_stdlib
    hardware_adc
    hardware_pwm
)

pico_add_extra_outputs(Practica11)
