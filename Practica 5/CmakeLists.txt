cmake_minimum_required(VERSION 3.13)

include($ENV{PICO_SDK_PATH}/external/pico_sdk_import.cmake)

project(led_gpio C CXX ASM)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

pico_sdk_init()

add_executable(led_gpio
    main.cpp
)

pico_add_extra_outputs(led_gpio)

target_link_libraries(led_gpio
    pico_stdlib
)
